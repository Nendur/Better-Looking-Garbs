on_host_change = {
	events = {
		blg.1 # New spawn ethnic diversity.
	}
}

on_birth = {
	events = {
		blg.2 # Set graphical culture from parents.
		#blg.7 # Freckles
	}
}

on_chronicle_start = {
	effect = {
		add_trait = i_used_to_be_a_player
		remove_trait = blgspawn
		any_character = { character_event = { id = blg.1 } } # Initial ethnic diversity.
	}
}

on_chronicle_owner_change = {
	effect = {
		add_trait = i_used_to_be_a_player
	}
}

on_society_bi_yearly_pulse = {
	effect = {
		#Some use immediate events and some delayed.
		character_event = { id = blg.3 days = 1 }
		character_event = { id = blg.3 days = 10 }
		character_event = { id = blg.3 days = 15 }
	}
}

on_quest_success = {
	effect = {
		remove_trait = blgquesting
	}
}

on_quest_failure = {
	effect = {
		remove_trait = blgquesting
	}
}
