	OR = {
		AND = {
			real_tier = duke
			independent = yes
			NOR = {
				blg_republic = yes
				blg_mercenary = yes
				blg_holy_order = yes
				blg_peasant = yes
			}
		}
		spouse_even_if_dead = {
			real_tier = duke
			independent = yes
			NOR = {
				blg_republic = yes
				blg_mercenary = yes
				blg_holy_order = yes
				blg_peasant = yes
			}
		}
	}

	OR = {
		culture = breton
		culture = reachmen
		culture = skaal
		culture = atmoran
		culture_group = yokudan_group
		culture = ayleid
		culture = orsimer
		culture = ashlander
		culture_group = kamal_group
		culture_group = tangmo_group
		culture_group = islander_group
	}
}
