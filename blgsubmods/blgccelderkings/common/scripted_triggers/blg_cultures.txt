### Culture triggers:

# Cultures with a lot of assimilating locals.
blg_assimilating_culture = {
	always = no
}

# Special cultures that don't mix with neighbours.
blg_culture_unmixed = {
	OR = {
		culture_group = daedra
		culture_group = undead
		culture_group = sload_group
		culture_group = dov
	}
}
