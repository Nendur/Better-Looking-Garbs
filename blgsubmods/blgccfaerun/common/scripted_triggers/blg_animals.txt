# Triggers for animal pets, mounts and familiars.

### Pets:

blg_animal_ape = {
	any_artifact = {
		is_artifact_equipped = yes
		artifact_type = ape_companion
	}
}

blg_animal_bat = {
	any_artifact = {
		is_artifact_equipped = yes
		artifact_type = bat_familiar
	}
}

blg_animal_boar = {
	OR = {
		any_artifact = {
			is_artifact_equipped = yes
			artifact_type = boar_companion
		}
		death_reason = death_accident_boar
		death_reason = death_execution_boar
	}
}

blg_animal_cat = {
	OR = {
		has_character_modifier = pet_cat
		has_character_modifier = child_cat
		any_artifact = {
			is_artifact_equipped = yes
			OR = {
				artifact_type = cat_familiar
				artifact_type = tressym_familiar
			}
		}
	}
}

blg_familiar_cat = {
	has_character_modifier = cat_familiar
	blg_visible_society = yes
}

blg_animal_dog = {
	OR = {
		has_character_modifier = pet_hunting_dog
		has_character_modifier = pet_house_dog
		has_character_modifier = pet_masterless_dog
		has_character_modifier = guide_dog
		has_character_modifier = child_dog
		any_artifact = {
			is_artifact_equipped = yes
			artifact_type = mastiff_companion
		}
	}
	NOT = { has_character_modifier = lost_dog }
}

blg_animal_eagle = {
	has_character_modifier = pet_hunting_eagle
}

blg_animal_falcon = {
	OR = {
		trait = falconer
		any_artifact = {
			is_artifact_equipped = yes
			OR = {
				artifact_type = hawk_familiar
				artifact_type = blood_hawk_companion
			}
		}
	}
}

blg_animal_fox = {
	always = no
}

blg_animal_frog = {

	}

blg_animal_hedgehog = {
	has_character_modifier = pet_hedgehog
}

blg_animal_imp = {

		is_artifact_equipped = yes
		OR = {
			artifact_type = imp_familiar
			artifact_type = quasit_familiar
		}
	}

blg_animal_monkey = {
	any_artifact = {
		is_artifact_equipped = yes
		artifact_type = flying_monkey_familiar
	}
}

blg_animal_otter = {
	OR = {
		has_character_modifier = pet_otter
		any_artifact = {
			is_artifact_equipped = yes
			artifact_type = weasel_familiar
		}
	}
}

blg_animal_owl = {
	OR = {
		has_character_modifier = pet_owl
		AND = {
			has_character_modifier = owl_familiar
			blg_visible_society = yes
		}
		any_artifact = {
			is_artifact_equipped = yes
			artifact_type = owl_familiar
		}
	}
}

blg_animal_parrot = {
	has_character_modifier = pet_parrot
}

blg_animal_rat = {
	OR = {
		has_character_modifier = pet_rat_dungeon
		has_character_modifier = rats_in_cell
		any_artifact = {
			is_artifact_equipped = yes
			OR = {
				artifact_type = rat_familiar
				artifact_type = cranium_rat_familiar
				artifact_type = giant_rat_companion
				artifact_type = mini_giant_space_hamster
			}
		}
	}
}

blg_animal_raven = {
	OR = {
		death_reason = death_murder_unknown_crows
		AND = {
			has_character_modifier = raven_familiar
			blg_visible_society = yes
		}
		any_artifact = {
			is_artifact_equipped = yes
			artifact_type = raven_familiar
		}
	}
}

blg_animal_snake = {
	OR = {
		death_reason = death_accident_snakes
		death_reason = death_murder_snakes
		death_reason = death_murder_unknown_snakes
		death_reason = death_execution_snake_pit
		has_character_modifier = pet_snake
		has_character_modifier = friend_of_snakes
		has_character_flag = diwali_snake_charmer
		any_artifact = {
			is_artifact_equipped = yes
			OR = {
				artifact_type = snake_familiar
				artifact_type = giant_poison_snake_companion
			}
		}
	}
}

blg_animal_vulture = {
	always = no
}

blg_animal_wolf = {
	OR = {
		AND = {
			has_character_modifier = wolf_familiar
			blg_visible_society = yes
		}
		any_artifact = {
			is_artifact_equipped = yes
			artifact_type = wolf_companion
		}
	}
}
