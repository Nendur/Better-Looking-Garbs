namespace = blg

# Ethnic diversity in random courtier spawns.
character_event = {
	id = blg.1
	hide_window = yes
	is_triggered_only = yes

	ai = yes
	min_age = 1
	trigger = {
		trait = blgspawn
	}
	immediate = {
		remove_trait = blgspawn
	}

	option = { # Leave unchanged / Ruler culture
		ai_chance = {
			factor = 6
		}
	}

	option = { # Local province
		ai_chance = { factor = 3 }
		trigger = {
			employer = {
				NOT = { character = no }
				is_landed = yes
				race = ROOT
			}
			OR = { dynasty = none num_of_dynasty_members <= 1 }
		}

		random_realm_province = {
			limit = { NOT = { province = no } }
			ROOT = {
				set_graphical_culture = PREV
			}
		}
	}

	option = { # Neighbour of local province
		ai_chance = { factor = 2 }
		trigger = {
			employer = {
				NOT = { character = no }
				is_landed = yes
				race = ROOT
			}
			OR = { dynasty = none num_of_dynasty_members <= 1 }
		}

		random_realm_province = {
			limit = { NOT = { province = no } }
			random_neighbor_province = {
				limit = { NOT = { province = no } }
				ROOT = {
					set_graphical_culture = PREV
				}
			}
		}
	}
}
