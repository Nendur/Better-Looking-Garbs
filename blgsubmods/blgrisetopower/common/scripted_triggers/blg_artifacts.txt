# Triggers for artifacts, items and weapons.

# Artifacts that can be used in combat.
blg_combat_artifacts = {
	OR = {
		blg_animal_eagle = yes
		any_artifact = {
			is_artifact_equipped = yes
			OR = {
				artifact_type = spear_of_destiny
				#artifact_type = mjolnir
			}
		}
	}
}
