### Terrain:

# Desert terrain.
blg_desert = {
	OR = {
		terrain = desert
		region = world_dorne
		region = world_qarth
	}
}

# Green Farmland terrain.
blg_farmlands = {
	OR = {
		terrain = farmlands
		AND = {
			OR = {
				terrain = plains
				terrain = hills
			}
			OR = {
				region = world_westeros_west
				region = world_westeros_east
				region = world_norvos
				region = world_qohor
			}
		}
	}
}

# Forested terrain.
blg_forest = {
	OR = {
		terrain = forest
		terrain = hills
		terrain = jungle
		terrain = arctic
		terrain = marsh
	}
}

# Mountain terrain.
blg_mountain = {
	OR = {
		terrain = mountain
		terrain = mountainpass
	}
}

# Steppe terrain.
blg_steppe = {
	terrain = plains
	OR = {
		region = world_dothraki_sea
		region = world_far_east
		region = world_shivering_sea
	}
}

# Snowy terrain.
blg_winter = {
	OR = {
		terrain = arctic
		region = world_the_wall
		region = world_beyond_the_wall
		has_province_modifier = asoiaf_mild_winter
		has_province_modifier = asoiaf_normal_winter
		has_province_modifier = asoiaf_harsh_winter
	}
}

# More common snowy terrain.
blg_severe_winter = {
	OR = {
		terrain = arctic
		has_province_modifier = asoiaf_normal_winter
		has_province_modifier = asoiaf_harsh_winter
	}
}
