### Hair rules:

# Balding further at high age.
blg_hair_balding = {
	is_female = no
	practical_age >= 70
}

# Has a shaved head.
blg_shaved = {
	OR = {
		AND = { #Norvoshi slaves are shaved
			liege = { culture = norvosi }
			blg_slave = yes
		}
		AND = { #Norvoshi women shave off all of their hair, though noblewomen don wigs
			OR = {
				culture = norvosi
				liege = { culture = norvosi }
			}
			is_female = yes
			dynasty = none
		}
	}
	NOT = { trait = bearded_priest }
}

# Tonsured hair rules.
blg_tonsure = {
	is_ruler = no
	trait = septon
	NOT = { trait = cynical }
}

# Not allowed to cut hair.
blg_hair_uncut = {
	trait = bearded_priest
}

# Tidy hair for rulers and married people.
blg_hair_chinese_tidy = {
	blg_hair_chinese = yes
	prisoner = no
	OR = {
		is_married = yes
		is_consort = yes
		is_landed = yes
		trait = groomed
	}
	NOT = { trait = uncouth }
}

# Tidy hair for married women and high rulers.
blg_hair_southernslavic_tidy = {
	is_female = yes
	OR = {
		is_married = yes
		higher_tier_than = duke
		trait = groomed
	}
	NOT = { trait = uncouth }
}

# Early Germanic kings wearing long hair.
blg_hair_germanic_ruler = {
	always = no
}

blg_hair_germanic_noruler = {
	always = no
}

# Tidy hair for married women and high rulers.
blg_hair_westernslavic_tidy = {
	is_female = yes
	OR = {
		is_married = yes
		higher_tier_than = count
		trait = groomed
	}
	NOT = { trait = uncouth }
}

### Beard rules:

# Always tries to grow a beard.
blg_beard_always = {
	trait = bearded_priest
}

# Has a long beard.
blg_beard_long = {
	trait = bearded_priest
}

# Has a shaved face.
blg_beard_short = {
	culture = norvosi
	NOT = { trait = bearded_priest }
}

# Cultures with uncommon growing of beards.
blg_beard_uncommon = {
	OR = {
		trait = slave
		trait = eunuch
		culture = nahuatl
		blg_beard_short = yes
	}
}
